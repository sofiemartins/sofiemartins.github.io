10 /* ***********************************************
11  * Simple output logging facility
12  * ***********************************************
13  * SYNOPIS:
14  *
15  * Messages outputted through the function
16  *
17  * int lprintf(char *name, int level, char*, ...);
18  *
19  * where name is an ID, level defines a log level,
20  * can be written to file streams according to a
21  * specified map. The map name => filename is
22  * defined (and can be modified) using the library
23  * functions:
24  *
25  * int logger_map(char *name, char *filename);
26  * int logger_unmap(char *name);
27  * int logger_unmap_all();
28  *
29  * All unmapped IDs are written to the logger
30  * stdout stream which can be set using:
31  *
32  * int logger_stdout(char *filename);
33  *
34  * (default = program stdout)
35  *
36  * ***********************************************/
37 
38 
39 /* map the ID name to the file with name filename
40  * filename can start with ">>" in which case the file
41  * is open in append mode
42  *
43  * Returns:
44  * 0 => success
45  * 1 => invalid name
46  * 2 => invalid filename
47  * 3 => cannot open new file
48  */
49 int logger_map(char *name, char *filename);
50 
51 /* unmap the ID name
52  * all subsequent calls to lprintf with this ID
53  * are mapped to the logger stdout
54  * If the function fails the mapping is unchanged
55  *
56  * Returns:
57  * 0 => success
58  * 1 => invalid name
59  */
60 int logger_unmap(char *name);
61 
62 /* reset the logger:
63  * all mappings are deleted
64  * all verbosity levels are deleted
65  * default verbosity level is set to 0
66  * logger stdout is mapped to application stdout
67  */
68 int logger_reset();
69 
70 /* set the logger stdout stream
71  * if filename=0 this function reset the stream to the
72  * program stdout
73  * NB: if filename is already open as a logger stream that stream become
74  * the logger stdout and all previously names mapped to that stream now write
75  * to stdout. If stdout is then changed all those names are written to a different
76  * stream.
77  *
78  * return codes:
79  * 0 => success
80  * 1 => failed to open new file (old logger stdout remains unchanged)
81  */
82 int logger_stdout(char *filename);
